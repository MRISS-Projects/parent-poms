31  @Service
32  public class JSoupParserFactory implements ClockinMailParserFactory<JsoupClockInParser> {
33  
34      @Override
35      public JsoupClockInParser getNewMailParser() {
36          return new JsoupClockInParser();
37      }
38  
39      @Override
40      public ClockinRecordDto getRecordsFromContent(JsoupClockInParser parser) {
41          List<Element> elements = parser.getElements();
42          ClockinRecordDto dto = getDtoFromElements(elements);
43          return dto;
44      }
45  
46      private ClockinRecordDto getDtoFromElements(List<Element> elements) {
47          ClockinRecordDtocessorservice/api/dto/ClockinRecordDto.html#ClockinRecordDto">ClockinRecordDto dto = new ClockinRecordDto();
48          ClockinEntry entry = null;
49          for (Iterator iterator = elements.iterator(); iterator.hasNext();) {
50              Element element = (Element) iterator.next();
51              if (element.tag() == Tag.valueOf("h3")) {
52                  dto.setUser(parseUser(element.ownText()));
53              } else if (element.tag() == Tag.valueOf("p")) {
54                  entry = new ClockinEntry();
55                  entry.setDate(parseDate(element.ownText()));
56                  dto.addEntry(entry);
57              } else if (element.tag() == Tag.valueOf("table")) {
58                  Elements records = getRecords(element);
59                  SingleRecord singleRecord = null;
60                  for (int i = 0; i < records.size(); i++) {
61                      String value = records.get(i).getElementsByTag("td").get(1).ownText().trim();
62                      if (i % 2 == 0) {
63                          singleRecord = new SingleRecord();
64                          singleRecord.setClockIn(value);
65                          entry.addRecord(singleRecord);
66                      } else {
67                          singleRecord.setClockOut(value);
68                      }
69                  }
70              }
71          }
72          return dto;
73      }
74  
75      private Elements getRecords(Element element) {
76          Elements elements = element.getElementsByTag("tbody");
77          Element tbody = elements.get(0);
78          return tbody.getElementsByTag("tr");
79      }
80  
81      private String parseDate(String ownText) {
82          return ownText.substring(JsoupClockInParser.DATE_MARKER.length() + 1);
83      }
84  
85      private String parseUser(String ownText) {
86          return ownText.substring(JsoupClockInParser.USER_MARKER.length() + 1);
87      }

89  }

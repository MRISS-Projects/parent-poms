16  @Service
17  public class JSoupParserFactory implements ClockinMailParserFactory<JsoupClockInParser> {
18  
19      @Override
20      public JsoupClockInParser getNewMailParser() {
21          return new JsoupClockInParser();
22      }
23  
24      @Override
25      public ClockinRecordDto getRecordsFromContent(JsoupClockInParser parser) {
26          List<Element> elements = parser.getElements();
27          ClockinRecordDto dto = getDtoFromElements(elements);
28          return dto;
29      }
30  
31      private ClockinRecordDto getDtoFromElements(List<Element> elements) {
32          ClockinRecordDto dto = new ClockinRecordDto();
33          ClockinEntry entry = null;
34          for (Iterator iterator = elements.iterator(); iterator.hasNext();) {
35              Element element = (Element) iterator.next();
36              if (element.tag() == Tag.valueOf("h3")) {
37                  dto.setUser(parseUser(element.ownText()));
38              } else if (element.tag() == Tag.valueOf("p")) {
39                  entry = new ClockinEntry();
40                  entry.setDate(parseDate(element.ownText()));
41                  dto.addEntry(entry);
42              } else if (element.tag() == Tag.valueOf("table")) {
43                  Elements records = getRecords(element);
44                  SingleRecord singleRecord = null;
45                  for (int i = 0; i < records.size(); i++) {
46                      String value = records.get(i).getElementsByTag("td").get(1).ownText().trim();
47                      if (i % 2 == 0) {
48                          singleRecord = new SingleRecord();
49                          singleRecord.setClockIn(value);
50                          entry.addRecord(singleRecord);
51                      } else {
52                          singleRecord.setClockOut(value);
53                      }
54                  }
55              }
56          }
57          return dto;
58      }
59  
60      private Elements getRecords(Element element) {
61          Elements elements = element.getElementsByTag("tbody");
62          Element tbody = elements.get(0);
63          return tbody.getElementsByTag("tr");
64      }
65  
66      private String parseDate(String ownText) {
67          return ownText.substring(JsoupClockInParser.DATE_MARKER.length() + 1);
68      }
69  
70      private String parseUser(String ownText) {
71          return ownText.substring(JsoupClockInParser.USER_MARKER.length() + 1);
72      }
73  
74  }
